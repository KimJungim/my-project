JavaScript 기능

버튼 컴포넌트에서 사용할 수 있는 자바스크립트 유틸리티 함수입니다. 주로 버튼의 로딩 상태나 비활성화 상태를 제어하는 데 사용됩니다.

버튼 로딩 상태 제어

<script>
  /**
   * 버튼 로딩 상태를 설정하는 함수
   * @param {HTMLElement} button - 대상 버튼 요소
   * @param {boolean} isLoading - 로딩 상태 여부
   * @param {string} loadingText - 로딩 중 표시할 텍스트 (기본값: '처리 중...')
   * @param {string} originalText - 원래 버튼에 표시할 텍스트 (제공되지 않으면 버튼의 현재 텍스트 사용)
   */
  function setButtonLoading(button, isLoading, loadingText = '처리 중...', originalText) {
    const btnText = originalText || button.textContent.trim();
    
    if (isLoading) {
      // 로딩 상태로 변경
      button.classList.add('btn--loading');
      button.setAttribute('disabled', true);
      button.innerHTML = `<span class="btn__spinner"></span> ${loadingText}`;
    } else {
      // 기본 상태로 복원
      button.classList.remove('btn--loading');
      button.removeAttribute('disabled');
      button.textContent = btnText;
    }
  }
  
  /**
   * 사용 예시:
   */
  document.addEventListener('DOMContentLoaded', function() {
    const demoBtn = document.getElementById('loadingDemoBtn');
    if (!demoBtn) return;
    
    demoBtn.addEventListener('click', function() {
      // 버튼을 로딩 상태로 변경
      setButtonLoading(this, true, '로딩 중...');
      
      // 3초 후 원래 상태로 복원 (실제 애플리케이션에서는 API 호출 완료 후 등의 시점에 호출)
      setTimeout(() => {
        setButtonLoading(this, false, '', '로딩 상태 테스트');
      }, 3000);
    });
  });
</script>
