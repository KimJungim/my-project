태그 타입별 비교

디자인 시스템에서 사용되는 네 가지 주요 태그 타입:

  • 기본 태그: 텍스트만 있는 단순한 태그입니다.
  • 버튼형 태그: 클릭 가능한 버튼형 태그로, 클릭 시 입력 필드에 태그 내용을 추가합니다.
  • 키워드 태그 (삭제 기능): 입력된 키워드를 표시하고 삭제할 수 있는 태그입니다.
  • 키워드 태그 (정적): 삭제 버튼이 없는 단순한 키워드 표시용 태그입니다.

JavaScript 기능

버튼형 태그 입력 기능

/**
 * 버튼형 태그 기능 초기화
 */
function initButtonTags() {
  const buttonTags = document.querySelectorAll('.tag--button');
  const tagInput = document.getElementById('tagInput');
  
  if (buttonTags.length === 0 || !tagInput) return;
  
  buttonTags.forEach(tag => {
    tag.addEventListener('click', function() {
      // 입력 필드에 태그 텍스트 설정
      tagInput.value = this.textContent;
    });
  });
}

// 초기화 호출
initButtonTags();

키워드 태그 삭제 기능

/**
 * 키워드 태그 삭제 기능 초기화
 */
function initTagCloseButtons() {
  const tagCloseButtons = document.querySelectorAll('.tag__close');
  
  if (tagCloseButtons.length === 0) return;
  
  tagCloseButtons.forEach(button => {
    button.addEventListener('click', handleTagDelete);
  });
}

/**
 * 태그 삭제 처리 함수
 * - 애니메이션 효과와 함께 태그 요소 제거
 */
function handleTagDelete(event) {
  const button = event.currentTarget;
  const tagElement = button.closest('.tag--keyword');
  
  if (tagElement) {
    // 애니메이션 효과 추가
    tagElement.style.opacity = '0';
    tagElement.style.transition = 'opacity 0.2s ease-out';
    
    // 애니메이션 후 요소 제거
    setTimeout(() => {
      tagElement.remove();
    }, 200);
  }
}

// 초기화 호출
initTagCloseButtons();
