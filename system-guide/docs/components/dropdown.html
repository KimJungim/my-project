JavaScript 기능

아래 스크립트는 드롭다운 컴포넌트의 동작을 위해 필요한 기능을 구현합니다. 주요 기능으로는 토글 기능, 항목 선택 기능, 외부 클릭 시 닫기 기능이 포함되어 있습니다.

참고: 아래 스크립트는 드롭다운의 기본적인 동작을 위해 필요합니다. 페이지에 드롭다운 컴포넌트가 포함된 경우 반드시 추가해 주세요. 
<!-- 접근성 유틸리티 스크립트 추가 -->
<script src="../../assets/scripts/common/a11y.js"></script>

<script>
  // a11y.js를 활용한 드롭다운 접근성 초기화
  document.addEventListener('DOMContentLoaded', function() {
    // 페이지의 모든 드롭다운 컴포넌트 초기화
    const dropdowns = document.querySelectorAll('.dropdown');
    dropdowns.forEach(dropdown => {
      App.a11y.initDropdown(dropdown);
    });
  });
</script>
