JavaScript 기능

아래 스크립트는 검색창 컴포넌트의 동작을 위해 필요한 기능을 구현합니다. 주요 기능으로는 버튼형 태그 클릭 시 검색 필드에 태그 텍스트를 입력하는 기능이 포함되어 있습니다.

참고: 아래 스크립트는 기능 검색창에서 태그 버튼을 사용할 때 필요합니다. 태그 버튼이 포함된 검색창을 사용하는 경우 반드시 추가해 주세요. 
<script>
  // 기능 검색창 태그 버튼 기능 초기화
  function initSearchboxTags() {
    const buttonTags = document.querySelectorAll('.tag--button');
    const tagInput = document.querySelector('.searchbox__input--feature');
    
    if (!buttonTags.length || !tagInput) return;
    
    buttonTags.forEach(tag => {
      tag.addEventListener('click', function() {
        // 입력 필드에 태그 텍스트 설정
        tagInput.value = this.textContent;
        
        // 필요한 경우 포커스 이동
        tagInput.focus();
      });
    });
  }
  
  // 페이지 로드 시 초기화
  document.addEventListener('DOMContentLoaded', function() {
    initSearchboxTags();
  });
</script>
